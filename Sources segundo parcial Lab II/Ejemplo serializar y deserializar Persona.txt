using System.Xml.Serialization;
using System.IO;

// Instancia la persona
Persona personita = new Persona("Ezequiel", "Mahafud");

// Crea un serializador para objetos de tipo Persona
XmlSerializer serializador = new XmlSerializer(typeof(Persona));

// Crea un escritor en el Escritorio
TextWriter escritor = new StreamWriter(Environment.GetFolderPath(Environment.SpecialFolder.Desktop) + "\\personas.xml", false, Encoding.UTF8);

// Serializa la persona
serializador.Serialize(escritor, personita);

// Cierra el serializador
escritor.Close();
